{{/* Partial: icon.html Renders an inline SVG icon by name (Lucide-style, stroke="currentColor"). Usage: {{ partial "icon" (dict "name" "compass" "size" "2rem" "class" "my-icon") }} File-based icons are loaded first from assets/icons/.svg. The file must use class="ICON_CLASS" as the placeholder. To add a new icon: just drop a .svg into assets/icons/ — no code changes needed. Inline fallback paths are kept for icons that have no file equivalent. */}} {{- $name := .name | default "" -}} {{- $size := .size | default "1.5rem" -}} {{- $class := .class | default "" -}} {{- $iconClass := printf "pkic-icon%s" (cond (ne $class "") (printf " %s" $class) "") -}} {{/* ── 1. File-based lookup: assets/icons/.svg ──────────────────────── */}} {{- $resource := resources.Get (printf "icons/%s.svg" $name) -}} {{- if $resource -}} {{- $content := $resource.Content | replaceRE `class="ICON_CLASS"` (printf `class="%s" width="%s" height="%s"` $iconClass $size $size) -}} {{ $content | safeHTML }} {{- else -}} {{/* ── 2. Inline fallback for icons without an SVG file ───────────────────── */}} {{- end -}}
